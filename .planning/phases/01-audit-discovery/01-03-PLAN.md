**Test structure:**

```typescript
describe('Regression Fixtures', () => {
  let chatKey: Uint8Array; // 32 random bytes, fixed for reproducibility
  let masterKey: CryptoKey;

  beforeAll(async () => {
    // Generate a deterministic chat key from known seed
    chatKey = new Uint8Array(32);
    // Fill with deterministic values for reproducibility
    for (let i = 0; i < 32; i++) chatKey[i] = i;

    // Generate a master CryptoKey for wrapping tests
    masterKey = await crypto.subtle.generateKey(
      { name: 'AES-GCM', length: 256 },
      true,
      ['encrypt', 'decrypt']
    );
  });

  describe('Format A: OM-header roundtrip', () => {
    it('encrypts and decrypts basic text', async () => {
      const plaintext = 'Hello, this is a test message for Format A';
      const ciphertext = await encryptWithChatKey(chatKey, plaintext);
      const decrypted = await decryptWithChatKey(chatKey, ciphertext);
      expect(decrypted).toBe(plaintext);
    });

    it('produces ciphertext starting with OM magic bytes', async () => {
      const ciphertext = await encryptWithChatKey(chatKey, 'test');
      const bytes = base64ToUint8Array(ciphertext);
      expect(bytes[0]).toBe(0x4F);
      expect(bytes[1]).toBe(0x4D);
    });

    it('includes correct fingerprint in header', async () => {
      const ciphertext = await encryptWithChatKey(chatKey, 'test');
      const bytes = base64ToUint8Array(ciphertext);
      const expectedFp = computeKeyFingerprint4Bytes(chatKey);
      expect(bytes.slice(2, 6)).toEqual(expectedFp);
    });
  });

  describe('Format B: Legacy format', () => {
    it('decrypts legacy format without OM header', async () => {
      // Manually construct a legacy ciphertext: [IV 12B][AES-GCM ciphertext]
      const plaintext = 'Legacy format test message';
      const encoder = new TextEncoder();
      const iv = crypto.getRandomValues(new Uint8Array(12));
      const rawKey = await crypto.subtle.importKey(
        'raw', chatKey, { name: 'AES-GCM' }, false, ['encrypt']
      );
      const encrypted = await crypto.subtle.encrypt(
        { name: 'AES-GCM', iv }, rawKey, encoder.encode(plaintext)
      );
      // Combine as legacy format: [IV][ciphertext]
      const combined = new Uint8Array(iv.length + encrypted.byteLength);
      combined.set(iv);
      combined.set(new Uint8Array(encrypted), iv.length);
      const legacyCiphertext = uint8ArrayToBase64(combined);

      // decryptWithChatKey should handle legacy format
      const decrypted = await decryptWithChatKey(chatKey, legacyCiphertext);
      expect(decrypted).toBe(plaintext);
    });
  });

  // ... Format C, D, error cases, edge cases
});
```

**For Format B legacy fixtures:** Manually construct `[IV 12B][AES-GCM ciphertext]` using `crypto.subtle.encrypt` directly, bypassing `encryptWithChatKey` (which adds the OM header). This simulates data written before the fingerprint header existed.

**Error cases:** Test that `decryptWithChatKey` with a wrong key:
- For Format A: should detect fingerprint mismatch before attempting AES-GCM
- For Format B: should fail with AES-GCM authentication error

**Edge cases:** Empty string, Unicode (including emoji like "Hello 👋🌍"), large message (generate 10KB string).
