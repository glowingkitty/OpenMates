    For each of the 35 uninstrumented handlers, add the ws_span_helper pattern. The handler function signature adds `user_otel_attrs: dict = None` as the last parameter. Then:

    ```python
    async def handle_xxx(websocket, manager, ..., payload, user_otel_attrs=None):
        _otel_span, _otel_token = None, None
        try:
            from backend.shared.python_utils.tracing.ws_span_helper import start_ws_handler_span, end_ws_handler_span
            _otel_span, _otel_token = start_ws_handler_span("xxx", user_id, payload, user_otel_attrs)
        except Exception:
            pass

        try:
            # ... existing handler body (UNCHANGED) ...
        except Exception as e:
            # ... existing error handling ...
            # Add: mark span as error
            if _otel_span is not None:
                try:
                    from backend.shared.python_utils.tracing.ws_span_helper import end_ws_handler_span
                    end_ws_handler_span(_otel_span, _otel_token, error=e)
                    _otel_span, _otel_token = None, None  # Prevent double-end in finally
                except Exception:
                    pass
            raise  # Re-raise if handler re-raises
        finally:
            if _otel_span is not None:
                try:
                    from backend.shared.python_utils.tracing.ws_span_helper import end_ws_handler_span
                    end_ws_handler_span(_otel_span, _otel_token)
                except Exception:
                    pass
    ```
